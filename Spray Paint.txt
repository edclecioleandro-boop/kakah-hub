-- LГӯngua padrГЈo ativa
local lingua_ativa = "chinese"
-- Prefixo para identificar o chat secreto (cada lГӯngua usa um prefixo Гәnico)
local prefixos = {
    chinese = "дёү",
    arabic = "Шҙ",
    symbols = "вЁі",
    portuguese = "рҹҮөрҹҮ№" -- Prefixo visual do ID de Portugal para o sistema reconhecer
}

-- FunГ§ГЈo para converter baseado na lГӯngua ativa
function convert(str, lang)
    str = str:lower()
    local dicionario = linguas[lang or lingua_ativa]
    for char, subst in pairs(dicionario) do
        str = str:gsub(char, subst)
    end
    return str
end

-- FunГ§ГЈo para reverter baseado na lГӯngua que foi recebida
function unconvert(str, lang)
    str = str:lower()
    local dicionario = linguas[lang]
    if not dicionario then return str end
    
    -- Inverte o dicionГЎrio para traduzir de volta
    local reverso = {}
    for k, v in pairs(dicionario) do
        reverso[v] = k
    end
    
    for subst, char in pairs(reverso) do
        str = str:gsub(subst, char)
    end
    return str
end

function chat(str)
    if game:GetService("TextChatService").ChatVersion == Enum.ChatVersion.LegacyChatService then
        game:GetService("ReplicatedStorage").DefaultChatSystemChatEvents.SayMessageRequest:FireServer(str,"All")
    else
        game:GetService("TextChatService").TextChannels.RBXGeneral:SendAsync(str)
    end
end

-- Detecta qual lГӯngua o outro jogador usou pelo prefixo e traduz de volta
function decodificar_mensagem(texto)
    for lang, prefixo in pairs(prefixos) do
        if texto:match(prefixo) then
            local limpo = texto:gsub(prefixo, "")
            return unconvert(limpo, lang), true
        end
    end
    return texto, false
end

chatScroller.DescendantAdded:Connect(function(msg)
    if msg:IsA("TextLabel") and msg.Name == "BodyText" then
        local label = msg
        local mensagem_original = msg.Text
        local traduzido, foi_secreto = decodificar_mensagem(mensagem_original)
        
        if foi_secreto then
            label.TextColor3 = Color3.new(1, 0.85098, 0) -- Amarelo para destacar
            label.Text = "    " .. traduzido
        end
    end
end)

if bubbleChat then
    bubbleChat.DescendantAdded:Connect(function(bubble)
        wait(0.5)
        if bubble:IsA("TextLabel") then
            local traduzido, foi_secreto = decodificar_mensagem(bubble.Text)
            if foi_secreto then
                bubble.Parent.BackgroundColor3 = Color3.new(1, 0.85098, 0)
                bubble.Text = traduzido
            end
        end
    end)
end
